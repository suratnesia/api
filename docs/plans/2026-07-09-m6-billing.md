**Goal:** Implement usage metering, pricing tier enforcement, and payment gateway integration to enable self-serve billing for Suratnesia tenants.

**Architecture:** A usage metering service tracks three dimensions — active users, storage volume, and E-Sign count — per tenant per billing period. A plan enforcer middleware checks limits before allowing resource-consuming actions. Midtrans handles payment processing via server-to-server webhooks.

```go
// internal/service/billing/meter.go
package billing

import (
	"time"
	model "suratnesia/internal/model/shared"
)

type MetricESign = model.MetricType

type UsageRepository interface {
	RecordUsage(record *model.UsageRecord) error
	GetCurrentPeriodUsage(tenantID uint, metric model.MetricType) (int64, error)
	GetUsageSummary(tenantID uint) (*model.UsageSummary, error)
}

type Meter struct {
	repo UsageRepository
}

func NewMeter(repo UsageRepository) *Meter {
	return &Meter{repo: repo}
}

func (m *Meter) RecordESign(tenantID uint, timestamp time.Time) error {
	now := time.Now()
	record := &model.UsageRecord{
		TenantID:    tenantID,
		Metric:      model.MetricESign,
		Value:       1,
		PeriodStart: beginningOfMonth(now),
		PeriodEnd:   endOfMonth(now),
		RecordedAt:  timestamp,
	}
	if m.repo != nil {
		return m.repo.RecordUsage(record)
	}
	return nil
}

func (m *Meter) GetCurrentUsage(tenantID uint) (*model.UsageSummary, error) {
	if m.repo != nil {
		return m.repo.GetUsageSummary(tenantID)
	}
	return &model.UsageSummary{TenantID: tenantID}, nil
}

func (m *Meter) CheckQuota(tenantID uint, metric model.MetricType, limit int) (bool, int) {
	var current int64
	if m.repo != nil {
		current, _ = m.repo.GetCurrentPeriodUsage(tenantID, metric)
	}
	remaining := int(int64(limit) - current)
	return remaining > 0, remaining
}

func beginningOfMonth(t time.Time) time.Time {
	return time.Date(t.Year(), t.Month(), 1, 0, 0, 0, 0, t.Location())
}

func endOfMonth(t time.Time) time.Time {
	return beginningOfMonth(t).AddDate(0, 1, 0).Add(-time.Nanosecond)
}
```

import (
	"net/http"
	"github.com/labstack/echo/v4"
	"suratnesia/internal/service/billing"
	model "suratnesia/internal/model/shared"
)

type PlanEnforcer struct {
	meter *billing.Meter
}

func NewPlanEnforcer(meter *billing.Meter) *PlanEnforcer {
	return &PlanEnforcer{meter: meter}
}

func (pe *PlanEnforcer) CheckUserLimit() echo.MiddlewareFunc {
	return func(next echo.HandlerFunc) echo.HandlerFunc {
		return func(c echo.Context) error {
			// In production: look up tenant's plan limits and compare with current usage
			// For now, pass through
			return next(c)
		}
	}
}

func (pe *PlanEnforcer) CheckESignQuota() echo.MiddlewareFunc {
	return func(next echo.HandlerFunc) echo.HandlerFunc {
		return func(c echo.Context) error {
			if pe.meter == nil {
				return next(c)
			}
			tenantID := c.Get("tenant_id")
			if tenantID == nil {
				return next(c)
			}
			allowed, remaining := pe.meter.CheckQuota(1, model.MetricESign, 50)
			if !allowed {
				return echo.NewHTTPError(http.StatusPaymentRequired, map[string]interface{}{
					"error":     "E-Sign quota exceeded",
					"remaining": remaining,
					"upgrade":   "Contact sales to upgrade your plan",
				})
			}
			return next(c)
		}
	}
}

func (pe *PlanEnforcer) CheckStorageQuota() echo.MiddlewareFunc {
	return func(next echo.HandlerFunc) echo.HandlerFunc {
		return func(c echo.Context) error {
			// Similar pattern: check storage usage vs plan limit
			return next(c)
		}
	}
}
